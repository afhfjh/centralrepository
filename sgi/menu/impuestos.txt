PROCEDURE proEjec_Opcion_xxx ( sbForma in VARCHAR2 )
IS
  sbNombForm VARCHAR2(500);
  fmIdenForm FORMMODULE;
  inSalvar   NUMBER(4,0);
  sbTemp		VARCHAR2(10);
  sbCad			VARCHAR2(10000);
  boTmp			BOOLEAN;
  inCont		NUMBER(5);

BEGIN
   -- Asigna el usuario
--   sbUsuario := get_application_property(USERNAME);
   --
   if (funVali_OpciMenu_Rol(UPPER(sbForma)) = TRUE) then
      :GLOBAL.SEL := upper(sbForma);
      sbNombForm := :GLOBAL.PATH || FunSeparador_Dir || lower(sbForma) || FunSeparador_Dir || lower(sbForma);
      fmIdenForm := Find_Form(:GLOBAL.SEL);
      IF Id_Null (fmIdenForm)then
      	
      			boTmp := TRUE;
      		IF (:global.open_forms IS NOT NULL) AND boTmp THEN
      				inCont := 1;
      				sbTemp := LOWER(sbForma);
      				WHILE sbTemp IS NOT NULL AND boTmp LOOP
      						sbCad := SUBSTR(:global.open_forms,inCont);
      						sbTemp := SUBSTR(sbCad,1,4);
      						IF sbTemp = LOWER(sbForma) THEN
      								boTmp := FALSE;
      						END IF;
      						inCont := inCont+4;
      				END LOOP;
      		END IF;
					IF boTmp THEN
					:global.open_forms := :global.open_forms||LOWER(sbForma);
					END IF;
				  -- Crear auditoria de acceso a forma
				  pkgFormAudi.proCrearAuditoria(sbForma,NULL,'S');
				  --
         OPEN_FORM(sbNombForm,ACTIVATE,SESSION);
     ELSE
         go_form(fmIdenForm);
      END IF;
   else
      inSalvar := funMens_Error_R('OPCION_NO_VALIDA',1020,get_application_property(USERNAME)||'|'||UPPER(sbForma));	
      raise form_trigger_failure;      
   end if;
	
END;
